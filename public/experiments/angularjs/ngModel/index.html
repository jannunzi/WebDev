<input ng-model="author" class="form-control"/>

<h3>Author: {{theAuthor}}</h3>
function NgModelController($scope) {
  $scope.setAuthor = setAuthor;

  function setAuthor(author) {
    $scope.theAuthor = author;
  }
}
