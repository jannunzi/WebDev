app.js

(function(){
  angular
    .module("WhiteBoardApp", ["ngRoute"])
    .config(function($routeProvider){
      $routeProvider
        .when("/", {
          templateUrl: "home.html"
        })
        .when("/profile", {
          templateUrl: "profile.html"
        })
        .when("/admin", {
          templateUrl: "admin.html"
        })
        .otherwise({
          redirectTo: "/"
        });
    });
})();
