Simple URL

Server

app.get('/api/hello', sayHello);

function sayHello(req, res) {
    console.log('Say Hello');
    res.send('<h1>Say Hello</h1>');
}

GET JSON

Simple URL

/api/json

Server

app.get('/api/json', function (req, res) {
    var course = {
        title: 'Java 101',
        seats: 23,
        start: new Date()
    };
    res.json(course);
});
